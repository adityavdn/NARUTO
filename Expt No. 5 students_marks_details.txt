#include<iostream>
#include<iomanip>
using namespace std;
class Student  //Creating a class named student
{ //Attributes
private: char name[30],grade;
int usn;
float mcmos,mcplus, mapp,
total,percentage;
public: //Constructor to initialize the variables
Student()
{ cout<<"\nIn Student Constructor";
name[0]='\0';
usn = 0;
grade='\0';
mcmos=mcplus=mapp=total=percentage= 0.0;
}
void getdata() //Function to get the data
{cout<<endl<<"Enter Name:";
cin>>name;
cout<<"Enter USN:";
cin>>usn;
cout<<endl<<"Enter marks in CMOS:";
cin>>mcmos;
cout<<endl<<"Enter marks in C++:";
cin>>mcplus;
cout<<endl<<"Enter marks in APP:";
cin>>mapp;
}
void calculate() //Function to calculate total and %ge
{ total = mcmos+mcplus+mapp;
percentage = total/300*100;
}
void showStudent()
{
    cout<<endl<<setprecision(5)<<name<<"\t"<<usn<<"\t"<<mcmos<<"\t"<<mcplus<< "\t "<<mapp<<"\t";
    cout<<setprecision(5)<<total<<"\t"<<percentage<<"\t "<<grade;
}
void findGrade() //Function to find grad
{
if(percentage>= 80) //Grading conditions
grade='A';
else if(percentage>= 70 && percentage < 80)
grade='B';
else if(percentage>= 60 && percentage < 70)
grade='C';
else
grade='F';
}
~Student()  //Destructor
{ cout<<"\n\nIn the Student Destructor\t";
cout<<"Year 2020-21"<<endl;
}
};//Student Class
int main()
{
int i,n;
cout<<"Enter the number of students : ";
cin>>n;
Student a[n]; //Array of objects
for(i=0;i<n;i++) //Loop to get and calculate the details
{
a[i].getdata();
a[i].calculate();
a[i].findGrade();
}
cout<<"\nName\tUSN\tCMOS\tCPP\tAdv ProcTotal\tPercentage\tGrade"<<endl;
for (i=0;i<n;i++) //Loop to display the students' details
{
a[i].showStudent();
}
return 0;
}
